  maxnames     = 4 ,

}

%% no first names
\renewcommand{\mkbibnamefirst}[1]{}

%% italicize titles
\DeclareFieldFormat
  [article]
  {title}{\textit{#1}}
\DeclareFieldFormat
  [book]
  {title}{\textit{#1}}

%% newline in bib entry
\newbibmacro*{bbx:parunit}{%
  \ifbibliography
    {\setunit{\bibpagerefpunct}\newblock
     \usebibmacro{pageref}%
     \clearlist{pageref}%
     \setunit{\adddot\par\nobreak}}
    {}}

%% year, not month+year
\renewbibmacro*{issue+date}{%
  \iffieldundef{year}
    {}
    {%
      {\printtext[parens]{\printfield{year}}}
    }%
  \newunit
}

% new bibliography driver, copied from nature.bbx then modified.
\DeclareBibliographyDriver{article}{%
  \usebibmacro{bibindex}%
  \usebibmacro{begentry}%
  \usebibmacro{author/translator+others}%
  \setunit{\labelnamepunct}\newblock
  \printlist{language}%
  %
  \newunit\newblock
  \usebibmacro{byauthor}%
  %
  \newunit\newblock
  \usebibmacro{journal+issuetitle}%
  %
  \newunit
  \usebibmacro{note+pages}%
  \newunit
  %
  %% \newunit\newblock
  %% \iffieldundef{pages}
  %%   {\usebibmacro{doi+eprint+url}}
  %%   {}%
  %%   %
  \setunit{\addspace}\newblock
  \usebibmacro{issue+date}%
  %
  \newunit\newblock
  \usebibmacro{bbx:parunit}\printfield{title}%
  \newblock
  \usebibmacro{finentry}%
}

%% this came from stackexchange... not a clue how it works!
\AtDataInput{%
  \csnumgdef{entrycount:\strfield{prefixnumber}}{%
    \csuse{entrycount:\strfield{prefixnumber}}+1}}

\DeclareFieldFormat{labelnumber}{\mkbibdesc{#1}}
\newrobustcmd*{\mkbibdesc}[1]{%
  \number\numexpr\csuse{entrycount:\strfield{prefixnumber}}+1-#1\relax}
